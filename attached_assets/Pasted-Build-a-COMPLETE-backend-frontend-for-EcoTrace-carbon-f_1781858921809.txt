CARBON CALCULATION ENGINE (REAL LOGIC)
═══════════════════════════════════════════════════════════

function calculateCarbon(type, value, unit, transport, veg) {
  // Travel: kg CO₂ per km
  const travelRates = {
    walking: 0,        // 0 kg CO₂/km
    cycling: 0,        // 0 kg CO₂/km
    bus: 0.05,         // 0.05 kg CO₂/km
    bike: 0.05,        // 0.05 kg CO₂/km
    car: 0.21,         // 0.21 kg CO₂/km
    flight: 0.25       // 0.25 kg CO₂/km
  };
  
  // Food: kg CO₂ per ₹100
  const foodRates = {
    veg: 0.5,          // ₹100 = 0.5 kg CO₂
    non_veg: 0.8       // ₹100 = 0.8 kg CO₂
  };
  
  // Electricity: kg CO₂ per ₹100
  const electricityRate = 0.12;  // ₹100 = 0.12 kg CO₂
  
  // Shopping: kg CO₂ per ₹100
  const shoppingRate = 0.4;      // ₹100 = 0.4 kg CO₂
  
  if (type === "travel") {
    return value * travelRates[transport]; // km × rate = kg CO₂
  }
  if (type === "food") {
    const rate = veg ? foodRates.veg : foodRates.non_veg;
    return (value / 100) * rate; // ₹ ÷ 100 × rate = kg CO₂
  }
  if (type === "electricity") {
    return (value / 100) * electricityRate;
  }
  if (type === "shopping") {
    return (value / 100) * shoppingRate;
  }
}

A. Google Location → Travel:
function classifyFromLocation(locationData) {
  const speed = locationData.speed; // km/h
  
  if (speed < 5) return { transport: "walking", type: "travel" };
  if (speed < 25) return { transport: "bike", type: "travel" };
  if (speed < 60) return { transport: "car", type: "travel" };
  return { transport: "flight", type: "travel" };
}

B. Fitness App → Walking/Cycling:
function classifyFromFitness(fitnessData) {
  const steps = fitnessData.steps;
  const walking_km = steps * 0.000762; // step = 0.762m
  return { 
    type: "travel", 
    value: walking_km, 
    unit: "km", 
    transport: "walking" 
  };
}

C. Bank Transactions → Food/Shopping:
function classifyFromBank(transaction) {
  const merchant = transaction.merchant.toLowerCase();
  const amount = transaction.amount; // ₹
  
  if (merchant.includes("zomato") || merchant.includes("swiggy")) {
    const veg = amount < 300; // <₹300 likely veg
    return { type: "food", value: amount, unit: "₹", veg: veg };
  }
  if (merchant.includes("amazon") || merchant.includes("flipkart")) {
    return { type: "shopping", value: amount, unit: "₹" };
  }
  if (merchant.includes("electricity")) {
    return { type: "electricity", value: amount, unit: "₹" };
  }
}

DATABASE SCHEMA (Prisma ORM)
═══════════════════════════════════════════════════════════

schema.prisma:

model User {
  id        String   @id @default(uuid())
  name      String
  email     String   @unique
  password  String
  created_at DateTime @default(now())
  activities Activity[]
  autoData   AutoData[]
  streaks    Streak[]
  aiChats    AIChat[]
}

model Activity {
  id             String   @id @default(uuid())
  user_id        String
  user           User     @relation(fields: [user_id], references: [id])
  type           String   // "travel", "food", "electricity", "shopping"
  value          Float    // distance, amount, units
  unit           String   // "km", "₹", "units", "item"
  transport      String?  // "car", "bike", "bus", "walking", "cycling" (for travel)
  veg            Boolean? // true/false (for food)
  carbon_emission Float   // kg CO₂
  auto_source    String?  // "manual", "lazy_mode", "vision", "gps", "fitness", "bank"
  confidence_score Float? // 0-1 (for AI vision)
  created_at     DateTime @default(now())
}

model AutoData {
  id              String   @id @default(uuid())
  user_id         String
  user            User     @relation(fields: [user_id], references: [id])
  source          String   // "google", "fitness", "bank", "ride_app"
  raw_data        String   // JSON string
  processed_carbon Float?
  created_at      DateTime @default(now())
}

model Streak {
  id              String   @id @default(uuid())
  user_id         String   @unique
  user            User     @relation(fields: [user_id], references: [id])
  current_streak  Int      @default(0)
  longest_streak  Int      @default(0)
  last_active_date DateTime
}

model AIChat {
  id         String   @id @default(uuid())
  user_id    String
  user       User     @relation(fields: [user_id], references: [id])
  message    String
  response   String
  created_at DateTime @default(now())
}
