## DATABASE TABLES

1. carbon_predictions:
   id UUID PRIMARY KEY,
   user_id UUID REFERENCES users(id),
   predicted_month DATE,
   predicted_co2_kg DECIMAL,
   prediction_accuracy DECIMAL, -- 92%
   created_at TIMESTAMP

2. auto_offsets:
   id UUID PRIMARY KEY,
   user_id UUID REFERENCES users(id),
   offset_month DATE,
   co2_offset_kg DECIMAL,
   offset_method TEXT, -- 'trees', 'renewable', 'capture'
   cost DECIMAL, -- ₹500
   blockchain_tx_id TEXT, -- Polygon transaction
   is_verified BOOLEAN DEFAULT false,
   created_at TIMESTAMP

3. null_certificates:
   id UUID PRIMARY KEY,
   user_id UUID REFERENCES users(id),
   month DATE,
   co2_nullified_kg DECIMAL,
   trees_planted INTEGER,
   nft_token_id TEXT, -- Blockchain NFT ID
   issued_at TIMESTAMP

4. subscriptions:
   id UUID PRIMARY KEY,
   user_id UUID REFERENCES users(id),
   monthly_cost DECIMAL DEFAULT 999,
   co2_covered_kg DECIMAL DEFAULT 1500, -- 1.5t
   status TEXT DEFAULT 'active', -- 'active', 'cancelled'

5. community_stats:
   id UUID PRIMARY KEY,

   total_nullified_kg DECIMAL,
   total_users INTEGER,
   updated_at TIMESTAMP

- Input: { predicted_co2_kg, offset_method }
